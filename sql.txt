CREATE TABLE MAIN
(
  ID Smallint NOT NULL,
  STATUS Integer DEFAULT 0,
  "COMMENT" Varchar(255)
);
UPDATE RDB$RELATION_FIELDS set RDB$DESCRIPTION = 'PRIMARY KEY'  where RDB$FIELD_NAME = 'ID' and RDB$RELATION_NAME = 'MAIN';
GRANT DELETE, INSERT, REFERENCES, SELECT, UPDATE
 ON MAIN TO  SYSDBA WITH GRANT OPTION;
